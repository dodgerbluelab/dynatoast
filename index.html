DynamicToast

- a lightweight toast system inspired by Apple's Dynamic Island

To get started, Add these two to the html meta tag:

<!-- DynamicToast CSS -->
<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/dodgerbluelab/dynatoast@latest/dynamic-toast.css">

<!-- DynaToast JavaScript -->
<script src="https://cdn.jsdelivr.net/gh/dodgerbluelab/dynatoast@latest/dynamic-toast.js"></script>

DynamicToast Options
text: 'Your toast message here' REQUIRED
type: 'success', 'info', 'warning', 'error' OPTIONAL
duration: 3000 (ms) OPTIONAL (Will fallback to 5000)

#1 DynamicToast


    DynamicToast({
        text: 'This is the DynamicToast',
        type: 'success',
        expanded: 'Please consider writing a thoughtful and clever expansion text.',
        duration: 5000
    })

#2 DynamicToast


    DynamicToast({
        text: 'Inspired by Apple. Built with joy and passion.',
        duration: 6000
    })

#3 DynamicToast


    DynamicToast({
        text: 'A new message was recieved',
        type: 'info',
        duration: 5000
    })

#4 DynamicToast


    DynamicToast({
        text: "Please be aware of your sharing you credentials",
        type: 'warning',
        duration: 2500
    })

#5 DynamicToast


    DynamicToast({
